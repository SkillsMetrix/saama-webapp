class Employee {
  constructor(empId, empName, empCity) {
    (this.empId = empId), (this.empName = empName), (this.empCity = empCity);
  }
}

let employees =
  JSON.parse(localStorage.getItem("empdata"))?.map(
    (emp) => new Employee(emp.empId, emp.empName, emp.empCity)
  ) || [];
let editIndex = -1;

function showEmp() {
  const list = document.getElementById("emplist");
  list.innerHTML = "";
  employees.forEach((emp, index) => {
    list.innerHTML += `
    <tr class="bg-gray-100 border-b">
    <td class="border border-x-gray-300 p-2 text-center">${emp.empId}</td>
     <td class="border border-x-gray-300 p-2 text-center">${emp.empName}</td>
      <td class="border border-x-gray-300 p-2 text-center">${emp.empCity}</td>
       <td class="border border-x-gray-300 p-2 text-center">
       <button  class="bg-yellow-600 px-2 py-1 rounded" onclick="editEmployee(${index})">Edit</button>
        <button onclick="confirmDelete(${index})" class="bg-red-600 px-2 py-1 rounded">Delete</button>
       
       </td>
    </tr>
      `;
  });
}
function saveToLocalStorage() {
  localStorage.setItem("empdata", JSON.stringify(employees));
}
function addorUpdateEmployee() {
  const empId = document.getElementById("empid").value;
  const empName = document.getElementById("empname").value;
  const empCity = document.getElementById("empcity").value;

  if (empId && empName && empCity) {
    if (editIndex === -1) {
      const empData = new Employee(empId, empName, empCity);
      employees.push(empData);
    } else {
      employees[editIndex] = { empId, empName, empCity };
      editIndex = -1;
      document.getElementById("btn").textContent = "Add Employee";
    }
    saveToLocalStorage();
    document.getElementById("empid").value = "";
    document.getElementById("empname").value = "";
    document.getElementById("empcity").value = "";

    showEmp();
  }
}

function editEmployee(index) {
  const emp = employees[index];
  document.getElementById("empid").value = emp.empId;
  document.getElementById("empname").value = emp.empName;
  document.getElementById("empcity").value = emp.empCity;
  document.getElementById("btn").textContent = "Update Employee";
  editIndex = index;
  console.log(editIndex);
}
function confirmDelete(index) {
  if (confirm("Are you sure to delete this record?")) deleteEmp(index);
}
function deleteEmp(index) {
  employees.splice(index, 1);
  saveToLocalStorage();
  showEmp();
}

document.addEventListener("DOMContentLoaded", showEmp);
